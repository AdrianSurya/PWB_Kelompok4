Digunakan untuk mengiterasi properti enumerable dari objek. 

    const person = {fname: "John", lname: "Doe", age: 25};
    for (let key in person) {
      console.log(key + ": " + person[key]);
    }
    // Output:
    // fname: John
    // lname: Doe
    // age: 25

Digunakan untuk mengiterasi elemen dalam objek iterable. 

    const fruits = ["apple", "banana", "cherry"];
    for (let fruit of fruits) {
      console.log(fruit);
    }
    // Output:
    // apple
    // banana
    // cherry

Menjalankan blok kode selama kondisi bernilai `true`. 

    let i = 0;
    while (i < 5) {
      console.log(i);
      i++;
    }
    // Output: 0, 1, 2, 3, 4

Digunakan untuk menghentikan loop lebih awal sebelum mencapai kondisi selesai. 

    for (let i = 0; i < 10; i++) {
      if (i === 5) break;
      console.log(i);
    }
    // Output: 0, 1, 2, 3, 4

Objek yang dapat diiterasi menggunakan `for...of`. 

    const iterable = "hello";
    for (let char of iterable) {
      console.log(char);
    }
    // Output: h, e, l, l, o

Koleksi nilai unik tanpa duplikasi. 

    const mySet = new Set(["apple", "banana", "cherry"]);
    mySet.add("date");
    console.log(mySet);
    // Output: Set(4) { "apple", "banana", "cherry", "date" }

Menambahkan, menghapus, dan memeriksa elemen dalam Set. 

    const mySet = new Set();
    mySet.add(1);
    mySet.add(2);
    console.log(mySet.has(1)); // true
    mySet.delete(1);
    console.log(mySet.has(1)); // false

Koleksi pasangan key-value yang terurut. 

    const myMap = new Map();
    myMap.set("name", "Alice");
    myMap.set("age", 25);
    console.log(myMap.get("name")); // Alice

Menambahkan, mengakses, dan menghapus pasangan key-value dalam Map. 

    const myMap = new Map();
    myMap.set("color", "blue");
    console.log(myMap.has("color")); // true
    myMap.delete("color");
    console.log(myMap.has("color")); // false

Operator `typeof` untuk memeriksa tipe data dan konversi tipe data. 

    console.log(typeof "hello"); // string
    console.log(Number("42"));   // 42
    console.log(String(42));     // "42"

Ekstraksi nilai dari array dan objek. 

    const [a, b] = [1, 2];
    const {name, age} = {name: "John", age: 30};
    console.log(name); // John
    console.log(age);  // 30

Operasi bitwise dengan operator seperti `&`, `|`, `^`. 

    const a = 5; // 0101
    const b = 3; // 0011
    console.log(a & b); // 1
    console.log(a | b); // 7

Digunakan untuk pencocokan pola dalam string. 

    const regex = /hello/i;
    const result = "Hello World".match(regex);
    console.log(result); // ["Hello"]

Menentukan urutan evaluasi operator dalam ekspresi. 

    console.log(2 + 3 * 4); // 14, karena * memiliki prioritas lebih tinggi daripada +

Digunakan untuk membuat objek baru dari constructor function. 

    function Person(name, age) {
      this.name = name;
      this.age = age;
    }
    const person1 = new Person("John", 30);
    console.log(person1.name); // John
